--- layout: post title: "Apache, Squid, Buildout" date: 2008-06-17 06:47:00 categories: description: Example buildout with apache and squid ---

Prerequisites 

  • Apache2 installed(can install with buildout like squid is done here)
  • Plone instance with buildout
[buildout]
parts = apacheconfig

[apacheconfig]
recipe = plone.recipe.apache:config
mainconfig = /path/to/included/apache/config
bind = 80
backends = 
    domain.org:127.0.0.1:3128
zope2_vhm_map = 
    domain.org:/domain

[squid-build]
recipe = plone.recipe.squid:build
url = http://www.squid-cache.org/Versions/v2/2.6/squid-2.6.STABLE18.tar.gz

[squid-instance]
recipe = plone.recipe.squid:instance
bind = 127.0.0.1:3128
backends = 127.0.0.1:8080
cache-size = 256
