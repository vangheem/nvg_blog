from boto.s3.connection import S3Connection
s3_key_id = '<id>'
s3_key_secret = '<secret>'
s3_conn = S3Connection(s3_key_id, s3_key_secret)
bucket = s3_conn.get_bucket('my-bucket')
key = bucket.get_key('filename.mp4')
expires = 30 * 24 * 60 * 60
url = key.generate_url(expires)
